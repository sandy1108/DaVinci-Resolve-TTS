# 字幕时间偏移问题分析（重要补充）

## 核心问题

### 你的分析完全正确！

**问题本质：**
- 🎤 **TTS API 返回的字幕时间** → 从 `00:00:00` 开始（相对于音频文件本身）
- 🎬 **实际插入位置** → 基于当前播放头（光标）位置
- ⚠️ **时间不匹配** → 字幕时间需要加上光标位置的偏移量

---

## 具体场景示例

### 场景 1：在时间线开头插入

```
播放头位置：00:00:00 (0 帧)
音频插入位置：00:00:00

TTS 返回的字幕：
  00:00:00,000 --> 00:00:02,000  第一句话
  00:00:02,000 --> 00:00:05,000  第二句话

实际应该的字幕位置：
  00:00:00,000 --> 00:00:02,000  第一句话  ✅ 正确
  00:00:02,000 --> 00:00:05,000  第二句话  ✅ 正确
```

**偏移量：0 帧** → 不需要调整

---

### 场景 2：在时间线中间插入（关键场景）

```
播放头位置：00:01:30,000 (2700 帧 @ 30fps)
音频插入位置：00:01:30,000

TTS 返回的字幕（相对于音频文件）：
  00:00:00,000 --> 00:00:02,000  第一句话
  00:00:02,000 --> 00:00:05,000  第二句话

❌ 错误：如果直接使用这些时间
  → 字幕会出现在 00:00:00 - 00:00:05
  → 但音频在 00:01:30 - 00:01:35
  → 字幕和音频不同步！

✅ 正确：需要加上偏移量
  00:01:30,000 --> 00:01:32,000  第一句话  (0 + 90秒)
  00:01:32,000 --> 00:01:35,000  第二句话  (2 + 90秒)
```

**偏移量：2700 帧（90 秒）** → 必须调整！

## 解决方案

### 方案 A：修改 SRT 文件的时间戳（推荐）

**核心思路：**
1. 获取音频插入位置（`start_frame`）
2. 解析 SRT 文件
3. 将所有时间戳加上偏移量
4. 生成新的 SRT 文件
5. 导入修改后的 SRT 文件

**优点：**
- ✅ 完全控制字幕时间
- ✅ 兼容性好（标准 SRT 格式）
- ✅ 可以同时处理增量替换

**实现代码：**

```python
def adjust_srt_timestamps(srt_path, offset_frames, frame_rate, output_path=None):
    """
    调整 SRT 文件中的所有时间戳，加上偏移量
    
    Args:
        srt_path: 原始 SRT 文件路径
        offset_frames: 偏移帧数（音频插入位置）
        frame_rate: 时间线帧率
        output_path: 输出文件路径（如果为 None，覆盖原文件）
    
    Returns:
        调整后的 SRT 文件路径
    """
    import re
    
    # 读取 SRT 文件
    with open(srt_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # 计算偏移量（秒）
    offset_seconds = offset_frames / frame_rate
    
    # SRT 时间戳格式：00:00:01,000 --> 00:00:03,000
    def adjust_timestamp(match):
        """调整单个时间戳"""
        time_str = match.group(0)
        
        # 解析时间：HH:MM:SS,mmm
        hours, minutes, seconds, milliseconds = re.match(
            r'(\d{2}):(\d{2}):(\d{2}),(\d{3})', 
            time_str
        ).groups()
        
        # 转换为总秒数
        total_seconds = (
            int(hours) * 3600 + 
            int(minutes) * 60 + 
            int(seconds) + 
            int(milliseconds) / 1000.0
        )
        
        # 加上偏移量
        new_total_seconds = total_seconds + offset_seconds
        
        # 转换回时间格式
        new_hours = int(new_total_seconds // 3600)
        new_minutes = int((new_total_seconds % 3600) // 60)
        new_seconds = int(new_total_seconds % 60)
        new_milliseconds = int((new_total_seconds % 1) * 1000)
        
        return f"{new_hours:02d}:{new_minutes:02d}:{new_seconds:02d},{new_milliseconds:03d}"
    
    # 替换所有时间戳
    # 匹配格式：00:00:01,000
    adjusted_content = re.sub(
        r'\d{2}:\d{2}:\d{2},\d{3}',
        adjust_timestamp,
        content
    )
    
    # 保存调整后的文件
    if output_path is None:
        output_path = srt_path
    
    with open(output_path, 'w', encoding='utf-8') as f:
        f.write(adjusted_content)
    
    print(f"字幕时间已调整，偏移量：{offset_seconds:.2f} 秒 ({offset_frames} 帧)")
    
    return output_path


def parse_srt_time_range(srt_path, frame_rate):
    """
    解析 SRT 文件，返回调整后的时间范围
    
    Returns:
        (start_frame, end_frame) - 已经包含偏移量的帧数
    """
    import re
    
    with open(srt_path, 'r', encoding='utf-8') as f:
        content = f.read()
    
    # 提取所有时间戳
    timestamps = re.findall(r'(\d{2}):(\d{2}):(\d{2}),(\d{3})', content)
    
    if not timestamps:
        return None, None
    
    def timestamp_to_frames(h, m, s, ms):
        """将时间戳转换为帧数"""
        total_seconds = int(h) * 3600 + int(m) * 60 + int(s) + int(ms) / 1000.0
        return int(total_seconds * frame_rate)
    
    # 第一个时间戳（开始时间）
    start_frame = timestamp_to_frames(*timestamps[0])
    
    # 最后一个时间戳（结束时间）
    end_frame = timestamp_to_frames(*timestamps[-1])
    
    return start_frame, end_frame
```

```python
def import_srt_to_timeline_with_offset(srt_path, insert_position_frame):
    """
    将字幕导入到时间线，支持时间偏移和增量替换
    
    Args:
        srt_path: SRT 文件路径
        insert_position_frame: 音频插入位置（帧数）
    
    Returns:
        True 表示成功，False 表示失败
    """
    timeline = current_project.GetCurrentTimeline()
    frame_rate = float(timeline.GetSetting("timelineFrameRate"))
    
    # ========== 步骤 1：调整 SRT 时间戳 ==========
    # 创建临时文件路径
    adjusted_srt_path = srt_path.replace('.srt', '_adjusted.srt')
    
    # 调整时间戳（加上偏移量）
    adjust_srt_timestamps(
        srt_path=srt_path,
        offset_frames=insert_position_frame,
        frame_rate=frame_rate,
        output_path=adjusted_srt_path
    )
    
    # ========== 步骤 2：解析调整后的时间范围 ==========
    start_frame, end_frame = parse_srt_time_range(adjusted_srt_path, frame_rate)
    
    if start_frame is None:
        print("无法解析 SRT 时间范围")
        return False
    
    print(f"字幕时间范围：{start_frame} - {end_frame} 帧")
    
    # ========== 步骤 3：删除重叠的旧字幕 ==========
    delete_overlapping_subtitles(timeline, start_frame, end_frame)
    
    # ========== 步骤 4：导入调整后的字幕 ==========
    media_pool = current_project.GetMediaPool()
    root_folder = media_pool.GetRootFolder()
    media_pool.SetCurrentFolder(root_folder)
    
    # 删除媒体池中的旧字幕文件（避免重复）
    file_name = os.path.basename(adjusted_srt_path)
    for clip in root_folder.GetClipList():
        if clip.GetName() == file_name:
            media_pool.DeleteClips([clip])
            break
    
    # 导入新字幕
    imported = media_pool.ImportMedia([adjusted_srt_path])
    
    if not imported:
        print(f"字幕导入失败: {adjusted_srt_path}")
        return False
    
    # ========== 步骤 5：添加到时间线 ==========
    success = media_pool.AppendToTimeline([imported[0]])
    
    if success:
        print(f"字幕已成功导入到时间线，位置：{insert_position_frame} 帧")
    else:
        print("字幕添加到时间线失败")
    
    # 清理临时文件（可选）
    # os.remove(adjusted_srt_path)
    
    return success


def delete_overlapping_subtitles(timeline, start_frame, end_frame):
    """
    删除与指定时间范围重叠的字幕片段
    
    Args:
        timeline: 时间线对象
        start_frame: 开始帧
        end_frame: 结束帧
    """
    sub_count = timeline.GetTrackCount("subtitle")
    total_deleted = 0
    
    for track_index in range(1, sub_count + 1):
        items = timeline.GetItemListInTrack("subtitle", track_index)
        
        if items:
            overlapping_items = []
            
            for item in items:
                item_start = item.GetStart()
                item_end = item.GetEnd()
                
                # 检查是否重叠
                # 重叠条件：NOT (item_end < start_frame OR item_start > end_frame)
                if not (item_end < start_frame or item_start > end_frame):
                    overlapping_items.append(item)
                    print(f"  发现重叠字幕：轨道 {track_index}, 范围 {item_start}-{item_end}")
            
            # 删除重叠的片段
            if overlapping_items:
                timeline.DeleteClips(overlapping_items)
                total_deleted += len(overlapping_items)
    
    if total_deleted > 0:
        print(f"已删除 {total_deleted} 个重叠的字幕片段")
    else:
        print("没有发现重叠的字幕片段")
```
